            dropZones.forEach(zone => {

                zone.addEventListener('drop', handleDrop);

            this.style.opacity = '1';
            draggedElement = null;

        let quizMatchingDraggedElement = null;

        function setupQuizMatchingGame() {
            const matchItems = document.querySelectorAll('.match-item');
            const matchZones = document.querySelectorAll('.match-zone');

            matchItems.forEach(item => {
                item.addEventListener('dragstart', handleQuizMatchDragStart);
                item.addEventListener('dragend', handleQuizMatchDragEnd);
            });

            matchZones.forEach(zone => {
                zone.addEventListener('dragover', handleQuizMatchDragOver);
                zone.addEventListener('drop', handleQuizMatchDrop);
                zone.addEventListener('dragenter', handleQuizMatchDragEnter);
                zone.addEventListener('dragleave', handleQuizMatchDragLeave);
            });
        }

        function handleQuizMatchDragStart(e) {
            quizMatchingDraggedElement = this;
            this.style.opacity = '0.5';
        }

        function handleQuizMatchDragEnd(e) {
            this.style.opacity = '1';
        }

        function handleQuizMatchDragOver(e) {
            e.preventDefault();
            return false;
        }

        function handleQuizMatchDragEnter(e) {
            this.style.backgroundColor = '#d5dbdb';
            this.style.borderColor = '#3498db';
        }

        function handleQuizMatchDragLeave(e) {
            this.style.backgroundColor = '#ecf0f1';
            this.style.borderColor = '#bdc3c7';
        }

        function handleQuizMatchDrop(e) {
            e.stopPropagation();
            this.style.backgroundColor = '#ecf0f1';
            this.style.borderColor = '#bdc3c7';

            if (quizMatchingDraggedElement && !this.querySelector('.match-item')) {
                this.appendChild(quizMatchingDraggedElement);
                quizMatchingDraggedElement.style.margin = '0';
                quizMatchingDraggedElement.style.width = '100%';
            }
            return false;
        }

        function setupQuizProgressTracking() {
            // Add event listeners to all quiz inputs
            const radioInputs = document.querySelectorAll('input[type="radio"]');
            const textInputs = document.querySelectorAll('#q3, #q8, #q12');
            
            radioInputs.forEach(input => {

        function updateQuizProgress() {
            let answeredQuestions = 0;
            
            // Check radio button questions
            for (let i = 1; i <= 15; i++) {
                if (i === 3 || i === 5 || i === 8 || i === 12) continue; // Skip fill-in and matching
                
                const radios = document.querySelectorAll(`input[name="q${i}"]`);
                for (let radio of radios) {
                    if (radio.checked) {
                        answeredQuestions++;
                        break;
                    }
                }
            }
            
            // Check fill-in questions
            const fillIns = ['q3', 'q8', 'q12'];
            fillIns.forEach(id => {
                const input = document.getElementById(id);
                if (input && input.value.trim() !== '') {
                    answeredQuestions++;
                }
            });
            
            // Check matching question
            const matchZones = document.querySelectorAll('.match-zone');
            let matchedItems = 0;
            matchZones.forEach(zone => {
                if (zone.querySelector('.match-item')) {
                    matchedItems++;
                }
            });
            if (matchedItems === 3) {
                answeredQuestions++;
            }
            
            // Update progress display
            const progressText = document.getElementById('quizProgress');
            const progressBar = document.getElementById('quizProgressBar');
            
            progressText.textContent = `${answeredQuestions}/15 –ø—Ä–∞—à–∞—ö–∞ –æ–¥–≥–æ–≤–æ—Ä–µ–Ω–∏`;
            progressBar.style.width = `${(answeredQuestions / 15) * 100}%`;
        }

        function checkQuiz() {
            let score = 0;
            let totalQuestions = 15;
            const results = [];
            
            // Check multiple choice questions
            const mcQuestions = [1, 2, 4, 6, 7, 9, 10, 11, 13, 14, 15];
            mcQuestions.forEach(qNum => {
                const selected = document.querySelector(`input[name="q${qNum}"]:checked`);
                if (selected && selected.value === quizAnswers[`q${qNum}`]) {
                    score++;
                    results.push({question: qNum, correct: true});
                } else {
                    results.push({question: qNum, correct: false, selected: selected ? selected.value : null});
                }
            });
            
            // Check fill-in questions
            const q3Answer = document.getElementById('q3').value.trim().toLowerCase();
            if (q3Answer === '—Ç—Ä–∞–Ω–∑–∏—Å—Ç–æ—Ä–∏' || q3Answer === '—Ç—Ä–∞–Ω–∑–∏—Å—Ç–æ—Ä') {
                score++;
                results.push({question: 3, correct: true});
            } else {
                results.push({question: 3, correct: false, answer: q3Answer});
            }
            
            const q8Answer = document.getElementById('q8').value.trim();
            if (q8Answer.toLowerCase().includes('apple') && q8Answer.toLowerCase().includes('i')) {
                score++;
                results.push({question: 8, correct: true});
            } else {
                results.push({question: 8, correct: false, answer: q8Answer});
            }
            
            const q12Answer = document.getElementById('q12').value.trim();
            if (q12Answer === '7') {
                score++;
                results.push({question: 12, correct: true});
            } else {
                results.push({question: 12, correct: false, answer: q12Answer});
            }
            
            // Check matching question
            const matchZones = document.querySelectorAll('.match-zone');
            let matchingCorrect = 0;
            matchZones.forEach(zone => {
                const placedItem = zone.querySelector('.match-item');
                if (placedItem) {
                    const itemType = placedItem.getAttribute('data-match');
                    const correctAnswer = zone.getAttribute('data-answer');
                    if (itemType === correctAnswer) {
                        matchingCorrect++;
                        zone.style.backgroundColor = '#d5f4e6';
                        zone.style.borderColor = '#27ae60';
                    } else {
                        zone.style.backgroundColor = '#fadbd8';
                        zone.style.borderColor = '#e74c3c';
                    }
                }
            });
            
            if (matchingCorrect === 3) {
                score++;
                results.push({question: 5, correct: true});
            } else {
                results.push({question: 5, correct: false});
            }
            
            // Display results
            displayQuizResults(score, totalQuestions, results);
        }

        function highlightWrongAnswers(results) {
            results.forEach(result => {
                const questionNum = result.question;
                
                if (!result.correct) {
                    // Find the question container
                    let questionContainer;
                    
                    if (questionNum === 3 || questionNum === 8 || questionNum === 12) {
                        // Fill-in questions
                        const input = document.getElementById(`q${questionNum}`);
                        if (input) {
                            input.style.borderColor = '#e74c3c';
                            input.style.borderWidth = '3px';
                            input.style.backgroundColor = '#fadbd8';
                            
                            // Find parent container and add red border
                            questionContainer = input.closest('div[style*="background: white"]');
                        }
                    } else if (questionNum === 5) {
                        // Matching question - already handled in checkQuiz function
                        const matchingContainer = document.querySelector('#generations').closest('div[style*="background: white"]');
                        if (matchingContainer) {
                            matchingContainer.style.borderColor = '#e74c3c';
                            matchingContainer.style.borderWidth = '3px';
                        }
                    } else {
                        // Multiple choice questions
                        const radios = document.querySelectorAll(`input[name="q${questionNum}"]`);
                        if (radios.length > 0) {
                            questionContainer = radios[0].closest('div[style*="background: white"]');
                            
                            // Highlight selected wrong answer
                            const selectedRadio = document.querySelector(`input[name="q${questionNum}"]:checked`);
                            if (selectedRadio) {
                                const label = selectedRadio.closest('label');
                                if (label) {
                                    label.style.backgroundColor = '#fadbd8';
                                    label.style.color = '#e74c3c';
                                    label.style.fontWeight = 'bold';
                                    label.style.padding = '5px';
                                    label.style.borderRadius = '5px';
                                    label.style.border = '2px solid #e74c3c';
                                }
                            }
                        }
                    }
                    
                    // Add red border to question container
                    if (questionContainer) {
                        questionContainer.style.borderColor = '#e74c3c';
                        questionContainer.style.borderWidth = '3px';
                        questionContainer.style.borderStyle = 'solid';
                    }
                } else {
                    // Highlight correct answers in green
                    let questionContainer;
                    
                    if (questionNum === 3 || questionNum === 8 || questionNum === 12) {
                        const input = document.getElementById(`q${questionNum}`);
                        if (input) {
                            input.style.borderColor = '#27ae60';
                            input.style.borderWidth = '3px';
                            input.style.backgroundColor = '#d5f4e6';
                            questionContainer = input.closest('div[style*="background: white"]');
                        }
                    } else if (questionNum !== 5) {
                        const selectedRadio = document.querySelector(`input[name="q${questionNum}"]:checked`);
                        if (selectedRadio) {
                            const label = selectedRadio.closest('label');
                            if (label) {
                                label.style.backgroundColor = '#d5f4e6';
                                label.style.color = '#27ae60';
                                label.style.fontWeight = 'bold';
                                label.style.padding = '5px';
                                label.style.borderRadius = '5px';
                                label.style.border = '2px solid #27ae60';
                            }
                            questionContainer = selectedRadio.closest('div[style*="background: white"]');
                        }
                    }
                    
                    if (questionContainer) {
                        questionContainer.style.borderColor = '#27ae60';
                        questionContainer.style.borderWidth = '3px';
                        questionContainer.style.borderStyle = 'solid';
                    }
                }
            });
